betaCheck :: Double -> Double -> Mealy Double (Double, Double)
betaCheck b r = (,) <$> xs' <*> ma'
  where
    xs' = depState (\a m -> a + b * m) (ma (1 - r))
    ma' = ma r >>> delay [0]
